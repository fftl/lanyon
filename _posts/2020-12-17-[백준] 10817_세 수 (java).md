- 10817_세 수 (java)

```java
import java.util.*;
import java.lang.*;
import java.io.*;

public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in); //Scanner 객체를 생성해 줌
        
        int a = sc.nextInt();            //각각 a b c정수를 입력 받음
        int b = sc.nextInt();
        int c = sc.nextInt();
        
        int[] numbers = {a, b, c};      //a b c를 인수로 가지는 numbers 배열을 선언
        
        Arrays.sort(numbers);           //numbers 배열을 정렬(기본정렬이 오름차순)
        
        System.out.println(numbers[1]); //세수를 정렬했으니 두번째 객체인 numbers[1]이 두번째로 큰수가 됨
    } 
}
```
